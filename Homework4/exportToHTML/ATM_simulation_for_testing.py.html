class BlankInputError(Exception):
    pass

def get_pin():
 try:
    pin_no = int(input("enter your pin code please "))
    if not pin_no:
        raise BlankInputError
 except BlankInputError as b:
     print(b)
 else:
     return pin_no


def check_pin(pin_no):
    file_object = 'clients.txt'
    with open(file_object, 'r+') as outfile:
        for line in outfile:
            if pin_no == line[-5:-1]:
                return True
    return False
